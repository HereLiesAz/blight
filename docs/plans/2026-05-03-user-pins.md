# User Pins & Auto-Hotspot Pins Implementation Plan (PR-1)

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add user-placed pins (long-press → category menu → tap to drop, emoji-glyph markers, localStorage-persisted) plus a derived "Hotspots (auto)" layer driven by `graffiti_score ≥ 0.8`. Phase 1 of three; spatial-pattern (PR-2) and OSM-derived (PR-3) auto-pins are deferred.

**Architecture:** Pure frontend addition to `index.html`. localStorage keys `user_pins` and `pin_categories` hold the data. Two new Leaflet layer groups (`userPinsLayer`, `hotspotLayer`) feed the existing `L.control.layers` overlay control. Long-press → Leaflet `contextmenu` event → popup with category buttons → `placePin(latlng, category)`. Auto-hotspot layer is repopulated by extending `applyFilters()`.

**Tech Stack:** Vanilla JS + Leaflet (existing). No Python changes. No new dependencies.

**Reference design:** [docs/plans/2026-05-03-user-pins-design.md](2026-05-03-user-pins-design.md)

### Step 1.2 — Default categories + storage helpers

Add a new `<script>` block at the end of the existing one (or extend the existing inline script block — pick whichever is less disruptive). Place these near the top of the script, before `let masterData = []`:

```javascript
const DEFAULT_PIN_CATEGORIES = [
    { name: "Target",       emoji: "🎯" },
    { name: "Scout",        emoji: "👁️" },
    { name: "Hit",          emoji: "🎨" },
    { name: "Photographed", emoji: "📷" },
    { name: "Done",         emoji: "✅" },
    { name: "Skip",         emoji: "🚫" },
];

function loadPinCategories() {
    try {
        const raw = localStorage.getItem('pin_categories');
        if (!raw) return [...DEFAULT_PIN_CATEGORIES];
        const parsed = JSON.parse(raw);
        return Array.isArray(parsed) && parsed.length ? parsed : [...DEFAULT_PIN_CATEGORIES];
    } catch { return [...DEFAULT_PIN_CATEGORIES]; }
}
function savePinCategories(cats) { localStorage.setItem('pin_categories', JSON.stringify(cats)); }

function loadUserPins() {
    try {
        const raw = localStorage.getItem('user_pins');
        if (!raw) return [];
        const parsed = JSON.parse(raw);
        return Array.isArray(parsed) ? parsed : [];
    } catch { return []; }
}
function saveUserPins(pins) { localStorage.setItem('user_pins', JSON.stringify(pins)); }

function pinCategoryFor(name) {
    return loadPinCategories().find(c => c.name === name) || { name, emoji: '📍' };
}
```

function renderUserPins() {
    userPinsLayer.clearLayers();
    const pins = loadUserPins();
    pins.forEach(pin => {
        const cat = pinCategoryFor(pin.category);
        const marker = L.marker([pin.lat, pin.lng], { icon: makePinIcon(cat.emoji) });
        marker.bindPopup(() => buildPinPopup(pin));

### Step 2.2 — Pin tap popup (placeholder for now; Task 3 makes it interactive)

```javascript
function buildPinPopup(pin) {
    const cat = pinCategoryFor(pin.category);
    const safeId = pin.id.replace(/'/g, "\\'");
    const cats = loadPinCategories();
    const options = cats.map(c =>
        `<option value="${c.name}" ${c.name === pin.category ? 'selected' : ''}>${c.emoji} ${c.name}</option>`
    ).join('');
    return `
        <div class="popup-content" style="min-width: 200px;">
            <h3>${cat.emoji} ${pin.category}</h3>
            <input type="text" placeholder="Title" value="${(pin.title||'').replace(/"/g,'&quot;')}" style="width:100%;padding:6px;background:#111;color:#0f0;border:1px solid #444;margin-bottom:6px;" onchange="updatePinField('${safeId}','title',this.value)">
            <textarea placeholder="Notes..." style="width:100%;height:60px;background:#111;color:#0f0;border:1px solid #444;padding:6px;font-family:monospace;" onchange="updatePinField('${safeId}','note',this.value)">${pin.note || ''}</textarea>
            <select onchange="updatePinField('${safeId}','category',this.value)" style="width:100%;padding:6px;background:#222;color:#eee;border:1px solid #555;margin-top:6px;">${options}</select>
            <button onclick="deletePin('${safeId}')" style="width:100%;padding:6px;background:#900;color:#fff;border:1px solid #f00;border-radius:4px;cursor:pointer;margin-top:6px;">🗑 Delete pin</button>
        </div>`;
}

window.updatePinField = function(id, field, value) {
    const pins = loadUserPins();
    const idx = pins.findIndex(p => p.id === id);
    if (idx === -1) return;
    pins[idx][field] = value;
    saveUserPins(pins);
    renderUserPins();  // re-render in case category changed (icon swap)
};

window.deletePin = function(id) {
    const pins = loadUserPins().filter(p => p.id !== id);
    saveUserPins(pins);
    renderUserPins();
    map.closePopup();
};
```

### Step 2.3 — Long-press detection + category menu

```javascript
function showCategoryMenu(latlng) {
    const cats = loadPinCategories();
    const buttons = cats.length > 0
        ? cats.map(c =>
            `<button onclick="placePin(${latlng.lat},${latlng.lng},'${c.name.replace(/'/g,"\\'")}'); map.closePopup();">${c.emoji} ${c.name}</button>`
          ).join('')
        : '<p style="color:#888">No categories yet — add one below.</p>';
    L.popup({ closeButton: true, className: '' })
        .setLatLng(latlng)
        .setContent(`
            <div class="pin-cat-menu">
                ${buttons}
                <button class="add-new" onclick="addCategoryFromMenu(${latlng.lat},${latlng.lng})">+ New category…</button>
                <button class="cancel" onclick="map.closePopup()">Cancel</button>
            </div>
        `)
        .openOn(map);
}

window.placePin = function(lat, lng, category) {
    const pins = loadUserPins();
    pins.push({
        id: `pin-${Date.now()}-${Math.floor(Math.random() * 1000)}`,
        lat, lng, category, title: '', note: '', created: new Date().toISOString(),
    });
    saveUserPins(pins);
    renderUserPins();
};

window.addCategoryFromMenu = function(lat, lng) {
    const emoji = prompt('Emoji for this category (e.g. 📍):', '📍');
    if (!emoji) return;
    const name = prompt('Name for this category:', '');
    if (!name) return;
    const cats = loadPinCategories();
    cats.push({ emoji, name });
    savePinCategories(cats);
    map.closePopup();
    showCategoryMenu(L.latLng(lat, lng));  // reopen with new category visible
};

// Long-press triggers `contextmenu` on iOS Safari + Android Chrome.
// Desktop right-click also triggers it (acceptable bonus).
map.on('contextmenu', e => {
    showCategoryMenu(e.latlng);
});

// Fallback: manual long-press detector for browsers that don't fire contextmenu on touch.
let _lpTimer = null, _lpStartXY = null;
map.getContainer().addEventListener('touchstart', e => {
    if (e.touches.length !== 1) return;
    _lpStartXY = { x: e.touches[0].clientX, y: e.touches[0].clientY };
    _lpTimer = setTimeout(() => {
        const t = e.touches[0];
        const point = map.containerPointToLatLng([
            t.clientX - map.getContainer().getBoundingClientRect().left,
            t.clientY - map.getContainer().getBoundingClientRect().top,
        ]);
        showCategoryMenu(point);
    }, 600);
}, { passive: true });
function _cancelLp() { if (_lpTimer) { clearTimeout(_lpTimer); _lpTimer = null; } }
['touchend', 'touchcancel', 'touchmove'].forEach(ev =>
    map.getContainer().addEventListener(ev, _cancelLp, { passive: true })
);

// Initial render at page load
renderUserPins();
```

```javascript
window.openPinCategories = function() {
    renderCategoryList();
    document.getElementById('pin-cat-modal').style.display = 'block';
};

function renderCategoryList() {
    const cats = loadPinCategories();
    const pins = loadUserPins();
    const list = document.getElementById('pin-cat-list');
    list.innerHTML = cats.map(c => {
        const inUse = pins.filter(p => p.category === c.name).length;
        const safeName = c.name.replace(/'/g, "\\'");
        return `<div class="cat-row">
            <span class="emoji">${c.emoji}</span>
            <span class="name">${c.name}${inUse ? ` <small style="color:#888">(${inUse} pin${inUse > 1 ? 's' : ''})</small>` : ''}</span>
            <button onclick="deleteCategory('${safeName}')">Delete</button>
        </div>`;
    }).join('') || '<p style="color:#888">No categories. Add one below.</p>';
}

window.addCategory = function() {
    const emoji = document.getElementById('new-cat-emoji').value.trim();
    const name = document.getElementById('new-cat-name').value.trim();
    if (!emoji || !name) { alert('Need both emoji and name.'); return; }
    const cats = loadPinCategories();
    if (cats.some(c => c.name === name)) { alert('That name is already used.'); return; }
    cats.push({ emoji, name });
    savePinCategories(cats);
    document.getElementById('new-cat-emoji').value = '';
    document.getElementById('new-cat-name').value = '';
    renderCategoryList();
};

window.deleteCategory = function(name) {
    const pins = loadUserPins();
    const inUse = pins.filter(p => p.category === name);
    if (inUse.length > 0) {
        const cats = loadPinCategories().filter(c => c.name !== name);
        if (cats.length === 0) { alert('Cannot delete the last category while pins exist. Add another first.'); return; }
        const fallback = cats[0].name;
        if (!confirm(`${inUse.length} pin${inUse.length > 1 ? 's are' : ' is'} in "${name}". Reassign to "${fallback}" and delete the category?`)) return;
        const updated = pins.map(p => p.category === name ? { ...p, category: fallback } : p);
        saveUserPins(updated);
        savePinCategories(cats);
        renderUserPins();
    } else {
        savePinCategories(loadPinCategories().filter(c => c.name !== name));
    }
    renderCategoryList();
};
```

## Task 4 — Auto-hotspot layer + QR export extension

**Files:** Modify `index.html`.

### Step 4.1 — Hotspot layer rendering

Add a function near `applyFilters`:

```javascript
const HOTSPOT_THRESHOLD = 0.8;
function renderHotspots() {
    hotspotLayer.clearLayers();
    masterData.forEach(item => {
        const s = item.prop.graffitiScore;
        if (s == null || s < HOTSPOT_THRESHOLD) return;
        const marker = L.marker([item.lat, item.lng], { icon: makePinIcon('🎨', 'hotspot') });
        marker.bindPopup(`
            <div class="popup-content">
                <h3>🎨 Auto-hotspot</h3>
                <p><b>Address:</b> ${item.prop.address}</p>
                <p><b>Score:</b> ${(s * 100).toFixed(0)}%</p>
            </div>`);
        hotspotLayer.addLayer(marker);
    });
}
```

Append `renderHotspots();` to the bottom of `applyFilters()` so the hotspot layer stays in sync with the underlying data on every re-render. Even though the hotspots themselves don't depend on filter state, they DO depend on `masterData`, which is repopulated by `fetchSectorData()` followed by `applyFilters()`.

### Step 4.2 — QR export extension

Find `window.generateQR`. Currently it serializes `txt_*` notes. Extend it to include user pins and categories:

```javascript
window.generateQR = function() {
    let stash = [];
    for (let i = 0; i < localStorage.length; i++) {
        const key = localStorage.key(i);
        if (key.startsWith('txt_')) stash.push({ a: key.replace('txt_', ''), n: getNote(key.replace('txt_', '')) });
    }
    const pins = loadUserPins();
    const cats = loadPinCategories();
    const payload = { stash, pins, cats };
    const json = JSON.stringify(payload);
    if (stash.length === 0 && pins.length === 0) return alert('No intel to beam.');
    document.getElementById('qrcode').innerHTML = '';
    new QRCode(document.getElementById("qrcode"), { text: json, width: 256, height: 256, correctLevel: QRCode.CorrectLevel.L });
    document.getElementById('qr-modal').style.display = 'block';
};
```

Note: this changes the QR payload schema from `[...notes...]` (an array) to `{ stash, pins, cats }` (an object). If you have an importer somewhere that reads QR codes, it'll need updating to match. Check whether one exists. If yes, update; if no, document the new schema in the design doc as already-correct future-importer expectation.
